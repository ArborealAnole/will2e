\newcommand\trim@spaces[1]{%
  \romannumeral-`\q\trim@trim@\noexpand#1Q Q%
}
\long\def\trim@trim@#1 Q{\trim@trim@@#1Q}
\long\def\trim@trim@@#1Q#2{#1}
\catcode`\Q=11

\newcommand\trim@spaces@noexp[1]{%
  \unexpanded\expandafter\expandafter\expandafter
    {\trim@spaces{#1}}%
}
      
% Brute force:
\newcommand\trim@spaces@in[1]{%
  \edef#1{\expandafter\trim@spaces@noexp\expandafter{#1}}%
}

\section{Use of the package}
This package provides three macros for the \LaTeX\ programmer:
\begin{itemize}
\item \cs{trim@spaces}\marg{text}\par 
Fully expandable, removes any surrounding spaces from \meta{text}.

\item \cs{trim@spaces@noexp}\marg{text}\par
Variant of the above that after full expansion results in the original unexpanded text stripped of any surrounding spaces.
(\textit{Requires $\varepsilon$-\TeX}.)

\item \cs{trim@spaces@in}\meta{macro}\par
Re-defines \meta{macro} as itself stripped of any surrounding spaces
(it is obviously not expandable).
\end{itemize}
These are best illustrated by example:
\begin{verbatim}
\def\foo{bar}
\edef\a{\trim@spaces{ a \foo ! }}
\edef\aa{\trim@spaces@noexp{ aa \foo ! }}
\def\b{ b \foo ! }
\trim@spaces@in\b
a:  [\detokenize\expandafter{\a}] \\
aa: [\detokenize\expandafter{\aa}] \\
b:  [\detokenize\expandafter{\b}]
\end{verbatim}
which produces:
\begin{quote}
\makeatletter
\def\foo{bar}
\edef\a{\trim@spaces{ a \foo ! }}
\edef\aa{\trim@spaces@noexp{ aa \foo ! }}
\def\b{ b \foo ! }
\trim@spaces@in\b
a:  [\detokenize\expandafter{\a}] \\
aa: [\detokenize\expandafter{\aa}] \\
b:  [\detokenize\expandafter{\b}]
\end{quote}
